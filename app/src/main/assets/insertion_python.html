# Main function for implementing insertion sort.
def insertionSort(arr):

    for i in range(1, len(arr)):

        key = arr[i]

        # Move all elements greater than the current ones.
        j = i-1
        while j >= 0 and key < arr[j] :
                arr[j + 1] = arr[j]
                j -= 1
        arr[j + 1] = key


arr = [4, 3, 5, 1, 2]

insertionSort(arr)
print("Sorted Array: \n")
for i in range(len(arr)):
    print (arr[i], end=" ")
