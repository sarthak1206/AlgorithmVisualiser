# Python program for implementation of Bubble Sort
def bubbleSort(arr):
    n = len(arr)

    for i in range(n):

        flag = 0
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
                flag = 1

        if flag == 0:
            break



# Driver code to test above
arr = [4, 3, 5, 1, 2]

bubbleSort(arr)

print("Sorted array: \n")
for i in range(len(arr)):
    print("%d" % arr[i], end=" ")
