<script>

// Main function for implementing insertion sort.
function insertionSort(arr, n)
{

    for (var i = 1; i < n; i++)
    {
        var key = arr[i];
        var j = i - 1;

        /* Move all elements greater than the current ones. */
        while (j >= 0 && arr[j] > key)
        {
            arr[j + 1] = arr[j];
            j = j - 1;
        }
        arr[j + 1] = key;
    }
}

// Function for printing an array. 
function printArray(arr, n)
{
    for (var i = 0; i < n; i++)
        document.write(arr[i] + " ");
    document.write("<br>");
}


let arr = [4, 3, 5, 1, 2 ];
let n = arr.length;

insertionSort(arr, n);

document.write("Sorted Array: ");
printArray(arr, n);


</script>
