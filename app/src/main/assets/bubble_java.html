/* Main Class for writing bubbleSort.*/
class BubbleSort {

    /* Main Function for implementing bubblesort. */
    void bubbleSort(int arr[])
    {
        int n = arr.length;
        for (int i = 0; i < n - 1; i++)
        {
            int flag = 0;
            for (int j = 0; j < n - i - 1; j++)
            {
                if (arr[j] > arr[j + 1]) {
                    int temp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = temp;
                    flag = 1;
                }
            }

            if(flag==0)
            {
                break;
            }
        }
    }

    /* Prints the array */
    void printArray(int arr[])
    {
        int n = arr.length;
        for (int i = 0; i < n; ++i)
            System.out.print(arr[i] + " ");
        System.out.println();
    }

    public static void main(String args[])
    {
        BubbleSort obj = new BubbleSort();
        int arr[] = { 4, 3, 5, 1, 2 };
        obj.bubbleSort(arr);
        System.out.println("Sorted array: \n");
        obj.printArray(arr);
    }
}
