<script>

function swap(arr, xp, yp)
{
    var temp = arr[xp];
    arr[xp] = arr[yp];
    arr[yp] = temp;
}

function bubbleSort( arr, n)
{
    var i, j;

    for (i = 0; i < n-1; i++)
    {
        var flag=0;
        for (j = 0; j < n-i-1; j++)
        {
            if (arr[j] > arr[j+1])
            {
                swap(arr,j,j+1);
                flag =1;
            }
        }

        if(flag===0)
        {
            break;
        }
    }
}

/* Function to print an array */
function printArray(arr, size)
{
    var i;
    for (i=0; i < size; i++)
        document.write(arr[i]+ " ");
    document.write("\n");
}


var arr = [4, 3, 5, 1, 2];
var n = 5;

bubbleSort(arr, n);
document.write("Sorted array: \n");
printArray(arr, n);


</script>
