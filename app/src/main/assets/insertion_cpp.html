#include <bits/stdc++.h>
using namespace std;

/* Main function for implementing insertion sort. */
void insertionSort(int arr[], int n)
{
    for(int i = 1; i < n; i++)
    {
        int key = arr[i];
        int j = i - 1;

        // Move all elements greater than the current ones.
        while (j >= 0 && arr[j] > key)
        {
            arr[j + 1] = arr[j];
            j = j - 1;
        }
        arr[j + 1] = key;
    }
}

/* Function for printing an array */
void printArray(int arr[], int n)
{
    for (int i = 0; i < n; i++)
        cout << arr[i] << " ";
    cout << endl;
}


int main()
{
    int arr[] = { 4, 3, 5, 1, 2};
    int N = 5;

    insertionSort(arr, N);

    cout<<"Sorted Array: "<<endl;
    printArray(arr, N);

    return 0;
}
