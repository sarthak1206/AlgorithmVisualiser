#include <bits/stdc++.h>

using namespace std;

// A main function of bubble sort
void bubbleSort(int arr[], int n)
{
    int i, j;
    for (i = 0; i < n - 1; i++)
    {
        int flag=0;
        for (j = 0; j < n - i - 1; j++)
        {
            if (arr[j] > arr[j + 1])
            {
                swap(arr[j], arr[j + 1]);
                flag=1;
            }
        }

        if(flag==0)
        {
            break;
        }
    }
}

// Function for printing an array. 
void printArray(int arr[], int n)
{
    int i;
    for (i=0; i<n; i++)
        cout << arr[i] << " ";
    cout << endl;
}

// Driver code
int main()
{
    int arr[] = { 4, 3, 5, 1, 2};
    int N = sizeof(arr) / sizeof(arr[0]);
    bubbleSort(arr, N);
    cout << "Sorted array: \n";
    printArray(arr, N);
    return 0;
}
