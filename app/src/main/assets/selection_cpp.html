#include <bits/stdc++.h>
using namespace std;


void selectionSort(int arr[], int n)
{
    int min_idx;

    for (int i = 0; i < n-1; i++)
    {
        // Finding the minimum element.
        min_idx = i;
        for (int j = i+1; j < n; j++)
        if (arr[j] < arr[min_idx])
            min_idx = j;

        if(min_idx!=i)
        {
            int temp = arr[min_idx];
            arr[min_idx] = arr[i];
            arr[i] = temp;
        }
    }
}

/* Function for printing an array. */
void printArray(int arr[], int size)
{
    for (int i=0; i < size; i++)
        cout << arr[i] << " ";
    cout << endl;
}


int main()
{
    int arr[] = {4, 3, 5, 1, 2};
    int n = 5;

    selectionSort(arr, n);

    cout << "Sorted array: \n";
    printArray(arr, n);

    return 0;
}
